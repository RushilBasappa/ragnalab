  <files>apps/uptime-kuma/docker-compose.yml</files>
  <action>
Create `apps/uptime-kuma/docker-compose.yml` with:

```yaml
# Uptime Kuma - Service Health Monitoring
# Access at: https://status.ragnalab.xyz

services:
  uptime-kuma:
    image: louislam/uptime-kuma:2
    container_name: uptime-kuma
    restart: unless-stopped
    volumes:
      - uptime-kuma-data:/app/data
      # Direct socket mount for container monitoring
      # (Socket proxy doesn't expose required endpoints for Uptime Kuma)
      - /var/run/docker.sock:/var/run/docker.sock:ro
    networks:
      - proxy
    labels:
      - "traefik.enable=true"
      - "traefik.http.routers.uptime-kuma.rule=Host(`status.ragnalab.xyz`)"
      - "traefik.http.routers.uptime-kuma.entrypoints=websecure"
      - "traefik.http.routers.uptime-kuma.tls=true"
      - "traefik.http.routers.uptime-kuma.tls.certresolver=letsencrypt"
      - "traefik.http.services.uptime-kuma.loadbalancer.server.port=3001"
      - "traefik.docker.network=proxy"
      # Backup label for stopping during backup
      - "docker-volume-backup.stop-during-backup=uptime-kuma"
    deploy:
      resources:
        limits:
          memory: 256M
          cpus: '0.5'
        reservations:
          memory: 128M
    logging:
      driver: json-file
      options:
        max-size: "10m"
        max-file: "3"

volumes:
  uptime-kuma-data:

networks:
  proxy:
    external: true
```

Key decisions:
- Direct Docker socket mount (read-only) for container monitoring - the socket proxy doesn't expose the endpoints Uptime Kuma needs
- Port 3001 is Uptime Kuma's default internal port
- backup stop label included for safe backups later
- Resource limits match research recommendations
