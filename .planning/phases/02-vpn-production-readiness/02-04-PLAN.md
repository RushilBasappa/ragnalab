  <action>
Run comprehensive automated verification of all Phase 2 success criteria:

```bash
#!/bin/bash
echo "=== Phase 2 Success Criteria Verification ==="
echo ""

PASS=0
FAIL=0

# Get IPs
TS_IP=$(tailscale ip -4)
LOCAL_IP=$(hostname -I | awk '{print $1}')

echo "Tailscale IP: $TS_IP"
echo "Local IP: $LOCAL_IP"
echo ""

# SC1: Production certificate
echo "SC1: Production Let's Encrypt certificate..."
CERT_ISSUER=$(echo | openssl s_client -connect $LOCAL_IP:443 -servername whoami.ragnalab.xyz 2>/dev/null | openssl x509 -noout -issuer 2>/dev/null)
if echo "$CERT_ISSUER" | grep -qi "let's encrypt\|R3\|R10\|R11\|E1"; then
  echo "  PASS: $CERT_ISSUER"
  ((PASS++))
else
  echo "  FAIL: Unexpected issuer: $CERT_ISSUER"
  ((FAIL++))
fi

# SC2: Dual access (local + VPN)
echo "SC2: Services accessible via local network..."
LOCAL_TEST=$(curl -sk --connect-timeout 5 "https://$LOCAL_IP" -H "Host: whoami.ragnalab.xyz" 2>&1 | head -1)
if echo "$LOCAL_TEST" | grep -qi "hostname\|ip\|method"; then
  echo "  PASS: Local network access works"
  ((PASS++))
else
  echo "  FAIL: Local network access failed: $LOCAL_TEST"
  ((FAIL++))
fi

echo "SC2b: Services accessible via Tailscale..."
VPN_TEST=$(curl -sk --connect-timeout 5 "https://$TS_IP" -H "Host: whoami.ragnalab.xyz" 2>&1 | head -1)
if echo "$VPN_TEST" | grep -qi "hostname\|ip\|method"; then
  echo "  PASS: Tailscale access works"
  ((PASS++))
else
  echo "  FAIL: Tailscale access failed: $VPN_TEST"
  ((FAIL++))
fi

# SC3: Tailscale persistence
echo "SC3: Tailscale service enabled..."
if systemctl is-enabled tailscaled >/dev/null 2>&1; then
  echo "  PASS: tailscaled service is enabled"
  ((PASS++))
else
  echo "  FAIL: tailscaled service not enabled"
  ((FAIL++))
fi

# SC4: HTTP to HTTPS redirect
echo "SC4: HTTP redirects to HTTPS..."
REDIRECT=$(curl -sk -o /dev/null -w "%{http_code}" --connect-timeout 5 "http://$LOCAL_IP" -H "Host: whoami.ragnalab.xyz")
if [ "$REDIRECT" = "301" ] || [ "$REDIRECT" = "308" ]; then
  echo "  PASS: HTTP returns $REDIRECT redirect"
  ((PASS++))
else
  echo "  FAIL: HTTP returns $REDIRECT (expected 301 or 308)"
  ((FAIL++))
fi

# SC5: Storage documented (informational)
echo "SC5: Storage architecture..."
DOCKER_ROOT=$(docker info 2>/dev/null | grep "Docker Root Dir" | cut -d: -f2 | xargs)
echo "  INFO: Docker root at $DOCKER_ROOT"
echo "  PASS: Storage architecture documented"
((PASS++))

echo ""
echo "=== Results: $PASS passed, $FAIL failed ==="
```
