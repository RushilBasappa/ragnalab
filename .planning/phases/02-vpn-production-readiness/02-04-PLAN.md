  <name>Task 2: Verify Tailscale state persistence</name>
  <files>None (verification only)</files>
  <action>
Test that Tailscale state persists across container restarts (VPN-02):

```bash
# Get current Tailscale IP
BEFORE_IP=$(docker exec tailscale tailscale ip -4)
echo "Before restart IP: $BEFORE_IP"

# Restart only the Tailscale container
docker restart tailscale

# Wait for container to be healthy
sleep 15

# Get IP after restart
AFTER_IP=$(docker exec tailscale tailscale ip -4)
echo "After restart IP: $AFTER_IP"

# Verify same IP (persistence working)
if [ "$BEFORE_IP" = "$AFTER_IP" ]; then
  echo "SUCCESS: Tailscale IP persisted across restart"
else
  echo "WARNING: Tailscale IP changed - state may not be persisting"
fi

# Verify still authenticated (no re-auth prompt)
docker exec tailscale tailscale status | head -5
```

Also verify the named volume exists:
```bash
docker volume inspect tailscale_state
```

<task type="auto">
  <name>Task 3: Automated success criteria verification</name>
  <files>None (verification only)</files>
  <action>
Run comprehensive automated verification of all Phase 2 success criteria:

```bash
#!/bin/bash
echo "=== Phase 2 Success Criteria Verification ==="
echo ""

PASS=0
FAIL=0

# Get IPs
TS_IP=$(docker exec tailscale tailscale ip -4)
LOCAL_IP=$(hostname -I | awk '{print $1}')

# SC1: Production certificate
echo "SC1: Production Let's Encrypt certificate..."
CERT_ISSUER=$(echo | openssl s_client -connect $TS_IP:443 -servername whoami.ragnalab.xyz 2>/dev/null | openssl x509 -noout -issuer 2>/dev/null)
if echo "$CERT_ISSUER" | grep -qi "let's encrypt\|R3\|R10\|R11\|E1"; then
  echo "  PASS: $CERT_ISSUER"
  ((PASS++))
else
  echo "  FAIL: Unexpected issuer: $CERT_ISSUER"
  ((FAIL++))
fi

# SC2: VPN-only access
echo "SC2: Services unreachable from local network..."
LOCAL_TEST=$(timeout 3 curl -sk "https://$LOCAL_IP:443" 2>&1)
if echo "$LOCAL_TEST" | grep -qE "(Connection refused|timed out|No route)"; then
  echo "  PASS: Local network access blocked"
  ((PASS++))
else
  echo "  FAIL: Services reachable from local network: $LOCAL_TEST"
  ((FAIL++))
fi

# SC3: Tailscale persistence (already tested in Task 2)
echo "SC3: Tailscale state persists..."
if docker volume inspect tailscale_state >/dev/null 2>&1; then
  echo "  PASS: tailscale_state volume exists"
  ((PASS++))
else
  echo "  FAIL: tailscale_state volume missing"
  ((FAIL++))
fi

# SC4: HTTP to HTTPS redirect
echo "SC4: HTTP redirects to HTTPS..."
REDIRECT=$(curl -sk -o /dev/null -w "%{http_code}" --connect-timeout 5 "http://$TS_IP" -H "Host: whoami.ragnalab.xyz")
if [ "$REDIRECT" = "301" ] || [ "$REDIRECT" = "308" ]; then
  echo "  PASS: HTTP returns $REDIRECT redirect"
  ((PASS++))
else
  echo "  FAIL: HTTP returns $REDIRECT (expected 301 or 308)"
  ((FAIL++))
fi

# SC5: Storage documented (informational)
echo "SC5: SSD storage validated..."
DOCKER_ROOT=$(docker info 2>/dev/null | grep "Docker Root Dir" | cut -d: -f2 | xargs)
echo "  INFO: Docker root at $DOCKER_ROOT"
echo "  PASS: Storage architecture documented"
((PASS++))

echo ""
echo "=== Results: $PASS passed, $FAIL failed ==="
```
